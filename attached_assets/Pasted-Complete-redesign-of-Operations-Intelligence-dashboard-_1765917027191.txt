Chart configuration (use Recharts):
```typescript
const data = [
  { stage: 'Manual', value: 5 },
  { stage: 'Tools', value: 18 },
  { stage: 'Core', value: 42 },
  { stage: 'Integration', value: 65 },
  { stage: 'Optimization', value: 83 },
  { stage: 'Autonomous', value: 94 }
];

<AreaChart data={data} height={240}>
  <defs>
    <linearGradient id="colorGradient" x1="0" y1="0" x2="0" y2="1">
      <stop offset="0%" stopColor="#00d4aa" stopOpacity={0.3}/>
      <stop offset="100%" stopColor="#00d4aa" stopOpacity={0.05}/>
    </linearGradient>
  </defs>
  
  <Area 
    type="monotone" 
    dataKey="value" 
    stroke="#00d4aa" 
    strokeWidth={2.5}
    fill="url(#colorGradient)"
    dot={{ fill: '#00d4aa', r: 4 }}
    activeDot={{ r: 6 }}
  />
  
  <Tooltip content={<CustomTooltip />} />
</AreaChart>
```

Custom Tooltip Component:
```typescript
const CustomTooltip = ({ active, payload }) => {
  if (!active || !payload?.[0]) return null;
  
  const tooltipContent = {
    5: {
      label: "MANUAL",
      desc: "Everything done by hand, reactive operations"
    },
    18: {
      label: "BASIC TOOLS",
      desc: "Basic software, still mostly manual"
    },
    42: {
      label: "AUTOMATED",
      desc: "Key workflows running automatically"
    },
    65: {
      label: "INTEGRATED",
      desc: "All systems connected, real-time sync"
    },
    83: {
      label: "OPTIMIZED",
      desc: "Zero bottlenecks, predictive operations"
    },
    94: {
      label: "AUTONOMOUS",
      desc: "Infrastructure handles operations, you handle strategy"
    }
  };
  
  const value = payload[0].value;
  const content = tooltipContent[value];
  
  return (
    <div style={{
      background: 'rgba(0, 0, 0, 0.92)',
      border: '2px solid #00d4aa',
      borderRadius: '8px',
      padding: '12px 16px',
      minWidth: '220px'
    }}>
      <div style={{
        fontSize: '20px',
        fontWeight: 700,
        color: '#00d4aa',
        marginBottom: '4px'
      }}>
        {value}%
      </div>
      <div style={{
        fontSize: '13px',
        fontWeight: 600,
        color: '#ffffff',
        marginBottom: '4px'
      }}>
        {content.label}
      </div>
      <div style={{
        fontSize: '12px',
        fontWeight: 400,
        color: 'rgba(255,255,255,0.8)',
        lineHeight: 1.5
      }}>
        {content.desc}
      </div>
    </div>
  );
};
```
